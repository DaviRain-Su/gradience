# GRA-213: MPP Handler 端到端测试

## Description

`payments/mpp-handler.ts` 有 823 行代码，处理多方支付（Multi-Party Payment）场景。需要端到端测试验证功能。

### MPP 支持的场景

1. **多 Judge 托管** - 多个评判者投票释放资金
2. **里程碑支付** - 分阶段释放
3. **收益分成** - 多方按比例分配
4. **条件释放** - 基于投票结果

## Implementation

### 1. 基础测试

```typescript
// tests/mpp-handler.e2e.test.ts
describe('MPP Handler E2E', () => {
    let mppHandler: MPPHandler;
    let connection: Connection;
    let payer: Keypair;
    
    beforeAll(async () => {
        connection = new Connection('https://api.devnet.solana.com');
        payer = await airdropTestWallet(connection);
        mppHandler = new MPPHandler(connection);
    });
    
    describe('Multi-Judge Escrow', () => {
        it('should create escrow with multiple judges', async () => {
            const payment = await mppHandler.createPayment({
                totalAmount: 1_000_000_000n,
                token: 'SOL',
                participants: [
                    { address: agent1.publicKey.toBase58(), shareBps: 9500, role: 'agent' },
                ],
                judges: [
                    { address: judge1.publicKey.toBase58(), weight: 1 },
                    { address: judge2.publicKey.toBase58(), weight: 1 },
                    { address: judge3.publicKey.toBase58(), weight: 1 },
                ],
                releaseConditions: {
                    type: 'voting',
                    threshold: 2, // 2/3 majority
                },
            });
            
            expect(payment.status).toBe('pending');
        });
        
        it('should release when majority votes', async () => {
            await mppHandler.vote(payment.paymentId, judge1, 'approve');
            await mppHandler.vote(payment.paymentId, judge2, 'approve');
            
            const updated = await mppHandler.getPayment(payment.paymentId);
            expect(updated.status).toBe('released');
        });
    });
    
    describe('Milestone Payments', () => {
        it('should release funds by milestone', async () => {
            const payment = await mppHandler.createPayment({
                totalAmount: 3_000_000_000n,
                releaseConditions: {
                    type: 'milestone',
                    milestones: [
                        { name: 'Phase 1', percentage: 30, deadline: Date.now() + 86400000 },
                        { name: 'Phase 2', percentage: 40, deadline: Date.now() + 172800000 },
                        { name: 'Phase 3', percentage: 30, deadline: Date.now() + 259200000 },
                    ],
                },
            });
            
            // Complete milestone 1
            await mppHandler.completeMilestone(payment.paymentId, 'Phase 1', judge1);
            
            const updated = await mppHandler.getPayment(payment.paymentId);
            expect(updated.participants[0].releasedAmount).toBe(900_000_000n); // 30%
        });
    });
    
    describe('Revenue Sharing', () => {
        it('should distribute to multiple participants', async () => {
            const payment = await mppHandler.createPayment({
                totalAmount: 1_000_000_000n,
                participants: [
                    { address: agent1, shareBps: 5000, role: 'agent' },      // 50%
                    { address: agent2, shareBps: 3000, role: 'contributor' }, // 30%
                    { address: provider, shareBps: 2000, role: 'provider' },  // 20%
                ],
            });
            
            await mppHandler.release(payment.paymentId);
            
            // Verify balances
            const agent1Balance = await connection.getBalance(agent1);
            expect(agent1Balance).toBeGreaterThanOrEqual(500_000_000);
        });
    });
});
```
