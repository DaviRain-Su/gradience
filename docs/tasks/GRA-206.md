# GRA-206: 实现 Private ER (PER) TEE 隐私执行

## Description

实现 MagicBlock Private Ephemeral Rollups，使用 Intel TDX TEE 进行隐私执行。用于 sealed-mode 任务的评判。

### 白皮书定义

> PER (Private ER): Sealed-mode execution via Intel TDX TEE. Enables confidential Judge evaluation—submissions remain encrypted throughout the scoring process.

### 使用场景

- **Sealed Submissions**: 提交内容在评判前保密
- **Confidential Judging**: 评判过程在 TEE 中进行
- **Private Results**: 只有 winner 公开，其他提交保密

## Implementation

### 1. PER 会话创建

```typescript
import { PrivateEphemeralRollup } from '@magicblock-labs/ephemeral-rollups-sdk';

export class PrivateEREngine {
    private per: PrivateEphemeralRollup;

    constructor(config: MagicBlockConfig) {
        if (!config.perEnabled) {
            throw new Error('PER not enabled in config');
        }
        this.per = new PrivateEphemeralRollup({
            endpoint: config.erEndpoint,
            apiKey: config.apiKey,
            teeProvider: 'intel-tdx',
        });
    }

    async createPrivateSession(
        accounts: PublicKey[],
        encryptionKey: Uint8Array
    ): Promise<PERSession> {
        const session = await this.per.createSession({
            accounts: accounts.map(a => a.toBase58()),
            encryptionKey,
            attestation: true,  // 要求 TEE 证明
        });
        return session;
    }
}
```

### 2. 加密提交处理

```typescript
interface SealedSubmission {
    taskId: string;
    submitter: string;
    encryptedResult: Uint8Array;
    commitment: string;  // hash(result || salt)
}

export async function processSealdSubmissions(
    engine: PrivateEREngine,
    task: Task,
    submissions: SealedSubmission[]
): Promise<JudgeResult> {
    // 1. 在 PER 中创建会话
    const session = await engine.createPrivateSession(
        [task.pda, task.escrowPda],
        task.encryptionKey
    );

    // 2. 验证 TEE attestation
    const attestation = await session.getAttestation();
    if (!verifyTEEAttestation(attestation)) {
        throw new Error('TEE attestation failed');
    }

    // 3. 在 TEE 中解密和评判
    const result = await session.executePrivate(async (teeContext) => {
        // 这段代码在 TEE 内执行
        const decrypted = submissions.map(s => ({
            submitter: s.submitter,
            result: teeContext.decrypt(s.encryptedResult),
        }));

        // 评判逻辑
        const winner = selectBestSubmission(decrypted);
        return { winner, score: calculateScore(winner) };
    });

    // 4. 只公开 winner，其他保密
    return {
        winner: result.winner,
        score: result.score,
        // 其他提交的内容不会离开 TEE
    };
}
```

    if (task.visibility === 'sealed') {
        // 使用 PER
