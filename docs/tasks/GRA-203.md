# GRA-203: 添加 MagicBlock SDK 依赖和基础配置

## Description

添加 MagicBlock 官方 SDK 依赖，配置基础连接和认证。这是所有 MagicBlock 功能的前置任务。

## 当前状态

`magicblock-enhancement.ts` 是空壳，没有真正的 SDK：

```typescript
// 当前：全是模拟
const delegateTx = 'delegate_tx_' + sessionId;  // 假的
```

## 目标

1. 添加 MagicBlock SDK 依赖
2. 配置 API 认证
3. 验证连接

## Implementation

### 1. 添加依赖

```bash
cd apps/agentm
pnpm add @magicblock-labs/ephemeral-rollups-sdk
```

### 2. 环境配置

```typescript
// .env
MAGICBLOCK_API_KEY=xxx
MAGICBLOCK_CLUSTER=devnet  // devnet | mainnet
MAGICBLOCK_ER_ENDPOINT=https://devnet.magicblock.app
```

### 3. 创建配置模块

```typescript
// src/main/settlement/magicblock-config.ts
export interface MagicBlockConfig {
    apiKey: string;
    cluster: 'devnet' | 'mainnet';
    erEndpoint: string;
    perEnabled: boolean;
    vrfEnabled: boolean;
}

export function loadMagicBlockConfig(): MagicBlockConfig {
    return {
        apiKey: process.env.MAGICBLOCK_API_KEY || '',
        cluster: (process.env.MAGICBLOCK_CLUSTER as 'devnet' | 'mainnet') || 'devnet',
        erEndpoint: process.env.MAGICBLOCK_ER_ENDPOINT || 'https://devnet.magicblock.app',
        perEnabled: process.env.MAGICBLOCK_PER_ENABLED === 'true',
        vrfEnabled: process.env.MAGICBLOCK_VRF_ENABLED === 'true',
    };
}
```

### 4. 连接验证

```typescript
import { EphemeralRollup } from '@magicblock-labs/ephemeral-rollups-sdk';

export async function verifyMagicBlockConnection(config: MagicBlockConfig): Promise<boolean> {
    try {
        const er = new EphemeralRollup({
            endpoint: config.erEndpoint,
            apiKey: config.apiKey,
        });
        await er.healthCheck();
        return true;
    } catch (error) {
        console.error('[MagicBlock] Connection failed:', error);
        return false;
    }
}
```
