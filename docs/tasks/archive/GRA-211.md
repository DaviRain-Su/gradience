## Implementation

### 1. 密钥管理

```typescript
// bridge/key-manager.ts
import { Keypair } from '@solana/web3.js';
import * as fs from 'fs';
import * as crypto from 'crypto';

export class BridgeKeyManager {
    private evaluatorKeypair: Keypair | null = null;
    
    async loadOrCreateEvaluatorKey(keyPath: string): Promise<Keypair> {
        if (fs.existsSync(keyPath)) {
            const encrypted = fs.readFileSync(keyPath);
            const decrypted = await this.decrypt(encrypted);
            this.evaluatorKeypair = Keypair.fromSecretKey(decrypted);
        } else {
            this.evaluatorKeypair = Keypair.generate();
            const encrypted = await this.encrypt(this.evaluatorKeypair.secretKey);
            fs.writeFileSync(keyPath, encrypted, { mode: 0o600 });
        }
        return this.evaluatorKeypair;
    }
    
    private async encrypt(data: Uint8Array): Promise<Buffer> {
        const password = process.env.DAEMON_KEY_PASSWORD;
        if (!password) throw new Error('DAEMON_KEY_PASSWORD not set');
        
        const salt = crypto.randomBytes(16);
        const key = crypto.scryptSync(password, salt, 32);
        const iv = crypto.randomBytes(16);
        const cipher = crypto.createCipheriv('aes-256-gcm', key, iv);
        
        const encrypted = Buffer.concat([
            cipher.update(Buffer.from(data)),
            cipher.final(),
        ]);
        const tag = cipher.getAuthTag();
        
        return Buffer.concat([salt, iv, tag, encrypted]);
    }
    
    private async decrypt(data: Buffer): Promise<Uint8Array> {
        const password = process.env.DAEMON_KEY_PASSWORD;
        if (!password) throw new Error('DAEMON_KEY_PASSWORD not set');
        
        const salt = data.subarray(0, 16);
        const iv = data.subarray(16, 32);
        const tag = data.subarray(32, 48);
        const encrypted = data.subarray(48);
        
        const key = crypto.scryptSync(password, salt, 32);
        const decipher = crypto.createDecipheriv('aes-256-gcm', key, iv);
        decipher.setAuthTag(tag);
        
        return new Uint8Array(Buffer.concat([
            decipher.update(encrypted),
            decipher.final(),
        ]));
    }
}
```

### 2. 真正的签名

```typescript
import { sign } from '@noble/ed25519';

private async signProof(hash: string): Promise<string> {
    if (!this.evaluatorKeypair) {
        throw new Error('Evaluator keypair not initialized');
    }
    
    const message = Buffer.from(hash, 'hex');
    const signature = await sign(
        message,
        this.evaluatorKeypair.secretKey.slice(0, 32)
    );
    
    return Buffer.from(signature).toString('hex');
}

// 验证签名
async verifySignature(hash: string, signature: string, pubkey: string): Promise<boolean> {
    const { verify } = await import('@noble/ed25519');
    return verify(
        Buffer.from(signature, 'hex'),
        Buffer.from(hash, 'hex'),
        Buffer.from(pubkey, 'hex')
    );
}
```

export async function createSettlementBridge(options: BridgeOptions = {}): Promise<SettlementBridge> {
