# GRA-212: A2A Router 完成 XMTP 集成

## Description

`a2a-router/router.ts` 已有 Nostr 集成，但 XMTP 标记为 "requires wallet signer at runtime"，需要完成。

### 当前状态

```typescript
// XMTP requires a wallet private key -- skip if not configured
if (this.options.enableXMTP) {
    logger.info('XMTP adapter enabled but requires wallet signer at runtime');
}
```

## Implementation

### 1. XMTP Adapter 集成

```typescript
import { XMTPAdapter, type XMTPAdapterOptions } from '@gradiences/xmtp-adapter';

async initialize(): Promise<void> {
    // ... Nostr init
    
    if (this.options.enableXMTP && this.walletSigner) {
        try {
            const xmtpOpts: XMTPAdapterOptions = {
                signer: this.walletSigner,
                env: this.options.xmtpEnv || 'production',
            };
            const xmtpAdapter = new XMTPAdapter(xmtpOpts);
            await xmtpAdapter.initialize();
            this.adapters.set('xmtp', xmtpAdapter);
            logger.info('XMTP adapter initialized');
        } catch (err) {
            logger.warn({ err }, 'XMTP adapter failed to initialize');
        }
    }
}
```

### 2. 钱包签名器接口

```typescript
// a2a-router/types.ts
export interface WalletSigner {
    getAddress(): Promise<string>;
    signMessage(message: Uint8Array): Promise<Uint8Array>;
}

// 从 OWS Wallet 创建签名器
export function createSignerFromOWSWallet(wallet: OWSWallet): WalletSigner {
    return {
        getAddress: async () => wallet.address,
        signMessage: async (message) => wallet.sign(message),
    };
}
```

### 3. 消息路由优先级

```typescript
// 根据消息类型选择协议
async send(message: A2AMessage): Promise<A2AResult> {
    const protocol = this.selectProtocol(message);
    
    switch (protocol) {
        case 'xmtp':
            // 直接消息使用 XMTP（E2E 加密）
            return this.sendViaXMTP(message);
        case 'nostr':
            // 广播/发现使用 Nostr
            return this.sendViaNostr(message);
    }
}

private selectProtocol(message: A2AMessage): ProtocolType {
    // 直接消息 → XMTP
    if (message.to && !message.broadcast) {
        if (this.adapters.has('xmtp')) {
            return 'xmtp';
        }
    }
    // 广播/发现 → Nostr
    return 'nostr';
}
```

### 4. 会话管理

```typescript
// XMTP 会话缓存
private xmtpConversations = new Map<string, Conversation>();

async getOrCreateConversation(peerAddress: string): Promise<Conversation> {
    if (this.xmtpConversations.has(peerAddress)) {
        return this.xmtpConversations.get(peerAddress)!;
    }
    
    const xmtp = this.adapters.get('xmtp') as XMTPAdapter;
    const conversation = await xmtp.client.conversations.newConversation(peerAddress);
    this.xmtpConversations.set(peerAddress, conversation);
    
    return conversation;
}
```
