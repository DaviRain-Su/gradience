# GRA-205: 实现 State Delegation 账户委托

## Description

实现 MagicBlock 的账户委托机制。账户需要先 delegate 到 ER，执行完后 undelegate 回 L1。

### 工作流程

```
L1 Account → delegate() → ER Account → execute() → commit() → undelegate() → L1 Account
```

## Implementation

### 1. Delegation 指令

```typescript
import { createDelegateInstruction, createUndelegateInstruction } from '@magicblock-labs/ephemeral-rollups-sdk';

export async function delegateAccountToER(
    connection: Connection,
    payer: Keypair,
    account: PublicKey,
    erProgramId: PublicKey,
): Promise<string> {
    const ix = createDelegateInstruction({
        payer: payer.publicKey,
        account,
        erProgramId,
    });

    const tx = new Transaction().add(ix);
    const sig = await sendAndConfirmTransaction(connection, tx, [payer]);
    return sig;
}

export async function undelegateAccountFromER(
    connection: Connection,
    payer: Keypair,
    account: PublicKey,
): Promise<string> {
    const ix = createUndelegateInstruction({
        payer: payer.publicKey,
        account,
    });

    const tx = new Transaction().add(ix);
    const sig = await sendAndConfirmTransaction(connection, tx, [payer]);
    return sig;
}
```

### 2. 批量委托

```typescript
export async function delegateMultipleAccounts(
    engine: MagicBlockExecutionEngine,
    accounts: PublicKey[],
): Promise<DelegationResult[]> {
    const results: DelegationResult[] = [];

    // 并行委托（有速率限制）
    const chunks = chunkArray(accounts, 10);
    for (const chunk of chunks) {
        const chunkResults = await Promise.all(
            chunk.map(async (account) => {
                try {
                    const sig = await engine.delegate(account);
                    return { account: account.toBase58(), success: true, signature: sig };
                } catch (error) {
                    return { account: account.toBase58(), success: false, error: String(error) };
                }
            }),
        );
        results.push(...chunkResults);
    }

    return results;
}
```

### 3. 委托状态追踪

```typescript
interface DelegationState {
    account: string;
    status: 'l1' | 'delegating' | 'er' | 'undelegating';
    delegatedAt?: number;
    erSessionId?: string;
    lastActivity?: number;
}

export class DelegationTracker {
    private states = new Map<string, DelegationState>();

    track(account: PublicKey): DelegationState {
        const key = account.toBase58();
        if (!this.states.has(key)) {
            this.states.set(key, { account: key, status: 'l1' });
        }
        return this.states.get(key)!;
    }

    updateStatus(account: PublicKey, status: DelegationState['status']): void {
        const state = this.track(account);
        state.status = status;
        state.lastActivity = Date.now();
    }

    getDelegatedAccounts(): DelegationState[] {
        return Array.from(this.states.values()).filter((s) => s.status === 'er');
    }
}
```
