### 2. 端到端测试

```typescript
// tests/e2e-revenue-distribution.test.ts
import { createRevenueDistributor, DistributionRequest } from '../revenue/distribution';

describe('Revenue Distribution E2E', () => {
    let distributor: RevenueDistributor;
    let testWallet: Keypair;
    
    beforeAll(async () => {
        distributor = createRevenueDistributor({
            rpcEndpoint: 'https://api.devnet.solana.com',
            chainHubProgramId: CHAIN_HUB_DEVNET_ID,
        });
        testWallet = await airdropTestWallet();
    });
    
    it('should distribute SOL correctly', async () => {
        const request: DistributionRequest = {
            paymentId: 'test-payment-1',
            taskId: 'test-task-1',
            agentAddress: new PublicKey('...'),
            judgeAddress: new PublicKey('...'),
            tokenMint: SystemProgram.programId, // Native SOL
            totalAmount: 1_000_000_000n, // 1 SOL
            escrowAccount: escrowPda,
            escrowAuthority: authorityPda,
        };
        
        const result = await distributor.distribute(request, testWallet);
        
        expect(result.status).toBe('confirmed');
        expect(result.breakdown.agent.amount).toBe(950_000_000n); // 95%
        expect(result.breakdown.judge.amount).toBe(30_000_000n);  // 3%
        expect(result.breakdown.protocol.amount).toBe(20_000_000n); // 2%
    });
    
    it('should distribute SPL tokens correctly', async () => {
        const request = {
            ...baseRequest,
            tokenMint: USDC_DEVNET_MINT,
            totalAmount: 1_000_000n, // 1 USDC (6 decimals)
        };
        
        const result = await distributor.distributeTokens(request, testWallet);
        
        expect(result.status).toBe('confirmed');
    });
});
```

### 4. 错误处理增强

```typescript
async distribute(request: DistributionRequest, signer: Signer): Promise<DistributionResult> {
    // 前置检查
    const escrowBalance = await this.connection.getBalance(request.escrowAccount);
    if (escrowBalance < Number(request.totalAmount)) {
        throw new DaemonError(
            ErrorCodes.INSUFFICIENT_FUNDS,
            `Escrow balance ${escrowBalance} < required ${request.totalAmount}`,
            400
        );
    }
    
    // 验证权限
    const escrowInfo = await this.connection.getAccountInfo(request.escrowAccount);
    if (!escrowInfo || !escrowInfo.owner.equals(this.config.chainHubProgramId)) {
        throw new DaemonError(
            ErrorCodes.INVALID_ACCOUNT,
            'Escrow account not owned by Chain Hub program',
            400
        );
```
