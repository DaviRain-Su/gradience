# GRA-208: 前端 UI 集成和执行模式切换

## Description

在 AgentM Web 前端添加 MagicBlock 执行模式的 UI，让用户可以选择使用 ER/PER 或直接 L1。

## Implementation

### 1. 执行模式选择器

```tsx
// components/settlement/ExecutionModeSelector.tsx
interface ExecutionMode {
    id: 'l1' | 'er' | 'per';
    name: string;
    description: string;
    available: boolean;
    recommended?: boolean;
}

const EXECUTION_MODES: ExecutionMode[] = [
    {
        id: 'l1',
        name: 'Solana L1',
        description: 'Direct settlement on Solana. ~400ms finality.',
        available: true,
    },
    {
        id: 'er',
        name: 'Ephemeral Rollup',
        description: 'MagicBlock ER. <10ms finality, 0 gas. Best for batch operations.',
        available: false, // 根据配置
        recommended: true,
    },
    {
        id: 'per',
        name: 'Private ER (TEE)',
        description: 'Confidential execution in Intel TDX. For sealed submissions.',
        available: false,
    },
];

export function ExecutionModeSelector({
    value,
    onChange,
    taskVisibility,
}: {
    value: ExecutionMode['id'];
    onChange: (mode: ExecutionMode['id']) => void;
    taskVisibility?: 'public' | 'sealed';
}) {
    // 如果是 sealed 任务，强制使用 PER
    const effectiveValue = taskVisibility === 'sealed' ? 'per' : value;

    return (
        <div className="execution-mode-selector">
            {EXECUTION_MODES.map((mode) => (
                <button
                    key={mode.id}
                    onClick={() => onChange(mode.id)}
                    disabled={!mode.available || (taskVisibility === 'sealed' && mode.id !== 'per')}
                    className={effectiveValue === mode.id ? 'selected' : ''}
                >
                    <span className="mode-name">{mode.name}</span>
                    <span className="mode-desc">{mode.description}</span>
                    {mode.recommended && <span className="badge">Recommended</span>}
                </button>
            ))}
        </div>
    );
}
```

### 2. MagicBlock 状态面板

```tsx
// components/settlement/MagicBlockStatus.tsx
export function MagicBlockStatus() {
    const { connection, sessions, metrics } = useMagicBlock();

    return (
        <div className="magicblock-status">
            <div className="connection-status">
                <span className={`indicator ${connection.status}`} />
                <span>MagicBlock: {connection.status}</span>
            </div>

            {sessions.length > 0 && (
                <div className="active-sessions">
                    <h4>Active ER Sessions</h4>
                    {sessions.map((s) => (
                        <div key={s.id} className="session-card">
                            <span>Session: {s.id.slice(0, 8)}...</span>
                            <span>State: {s.state}</span>
                            <span>Accounts: {s.accountCount}</span>
                        </div>
                    ))}
                </div>
            )}

            {metrics && (
                <div className="metrics">
                    <div>Ops: {metrics.operationsExecuted}</div>
                    <div>Avg Latency: {metrics.avgExecutionTimeMs}ms</div>
                    <div>Commits: {metrics.commitsToL1}</div>
                </div>
            )}
        </div>
    );
}
```

### 3. 任务创建时的模式选择

```tsx
// 在 TaskCreateForm 中集成
function TaskCreateForm() {
    const [visibility, setVisibility] = useState<'public' | 'sealed'>('public');
    const [executionMode, setExecutionMode] = useState<'l1' | 'er' | 'per'>('l1');

    return (
        <form>
            {/* 其他字段... */}

            <div className="field">
                <label>Task Visibility</label>
                <select value={visibility} onChange={(e) => setVisibility(e.target.value)}>
                    <option value="public">Public (all submissions visible)</option>
                    <option value="sealed">Sealed (encrypted until judged)</option>
                </select>
            </div>

            <div className="field">
                <label>Execution Mode</label>
                <ExecutionModeSelector value={executionMode} onChange={setExecutionMode} taskVisibility={visibility} />
            </div>

            {visibility === 'sealed' && (
                <div className="info-box">Sealed tasks require Private ER (TEE) for confidential judging.</div>
            )}
        </form>
    );
}
```
