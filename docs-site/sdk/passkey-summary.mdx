## 使用示例

### 创建 Passkey Wallet

```typescript
import { usePasskeyWallet } from './hooks/usePasskeyWallet';

function CreateAgent() {
  const { createWallet, isLoading, error } = usePasskeyWallet({
    rpId: 'agentm.gradience.io',
    rpName: 'Gradience AgentM',
    userId: 'user_abc123',
    userName: 'john@example.com',
  });

  const handleCreate = async () => {
    const credential = await createWallet({
      agentId: 'agent_xyz789',
      masterWalletAddress: '5Y3d...7xKp',
      derivationIndex: 0,
    });
    
    if (credential) {
      console.log('Created:', credential.credentialId);
    }
  };

  return (
    <button onClick={handleCreate} disabled={isLoading}>
      {isLoading ? 'Creating...' : 'Create Agent Wallet'}
    </button>
  );
}
```

### 恢复 Wallet

```typescript
import { PasskeyWalletRecovery } from './components/PasskeyWalletRecovery';

function RecoveryPage() {
  return (
    <PasskeyWalletRecovery
      rpId="agentm.gradience.io"
      rpName="Gradience AgentM"
      userId="user_abc123"
      userName="john@example.com"
      onRecovered={(wallet) => {
        console.log('Recovered:', wallet.agentId);
      }}
    />
  );
}
```

### 迁移旧数据

```typescript
import { usePasskeyMigration } from './lib/migration/localstorage-to-passkey';

function MigrationPage() {
  const { status, migrate, isLoading } = usePasskeyMigration({
    rpId: 'agentm.gradience.io',
    rpName: 'Gradience AgentM',
    userId: 'user_abc123',
    userName: 'john@example.com',
  });

  if (status?.recommendedAction === 'migrate') {
    return (
      <button onClick={migrate} disabled={isLoading}>
        迁移到 Passkey
      </button>
    );
  }
}
```

## 安全特性

| 特性 | 实现 |
|------|------|
| 私钥存储 | Secure Enclave / TPM (硬件级) |
| 加密算法 | AES-256-GCM |
| 密钥派生 | SHA-256 + 钱包地址 |
| 认证方式 | FaceID / TouchID / Windows Hello |
| 同步 | iCloud / Google (端到端加密) |
